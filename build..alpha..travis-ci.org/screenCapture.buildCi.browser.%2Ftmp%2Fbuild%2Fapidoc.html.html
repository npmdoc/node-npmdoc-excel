    function parseXlsx(path, sheet, cb) {
	if (typeof cb === 'undefined') {
		cb = sheet;
		sheet = '1';
	}
	extractFiles(path, sheet).then(function(files) {
		cb(null, extractData(files));
	},
	function(err) {
		cb(err);
	});
}
